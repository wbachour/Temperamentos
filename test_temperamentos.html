        <button type="button" onclick="calculateTemperament()">Calcular Resultado</button>

    <script>

        function calculateTemperament() {

            const groups = 40;

            const scores = {

                colerico: 0,

                melancolico: 0,

                sanguineo: 0,

                flematico: 0

            };


            for (let i = 1; i <= groups; i++) {

                const selected = document.querySelector(`input[name='group${i}']:checked`);

                if (selected) {

                    scores[selected.value]++;

                }

            }


            const maxTemperament = Object.keys(scores).reduce((a, b) => scores[a] > scores[b] ? a : b);


            document.getElementById('results').innerHTML = `Tu temperamento dominante es: <strong>${maxTemperament.toUpperCase()}</strong><br>

                <p>Colérico: ${scores.colerico} | Melancólico: ${scores.melancolico} | Sanguíneo: ${scores.sanguineo} | Flemático: ${scores.flematico}</p>`;

        }

    </script>
